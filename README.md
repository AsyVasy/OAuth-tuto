``` module.exports = {
	google: {
		clientID: '************-*********************************.apps.googleusercontent.com',
		clientSecret: '************************',
	},
	mongodb: {
		dbURI: 'mongodb://*******:****************@******.mlab.com:***/nn-oauth-test',
	},
	session: {
		cookieKey: '****************',
	},
};
```
